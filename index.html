Quick Start

# Clone the repository
git clone https://github.com/OpenGELab/GELab-Zero.git
cd GELab-Zero

# Install dependencies
pip install -r requirements.txt

# Download the 4B model
python download_model.py --model 4B

# Connect to your mobile device
python connect_device.py --device 

# Run inference
python inference.py --task "Open settings and enable dark mode"
